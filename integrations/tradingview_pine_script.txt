// MNQ Data Export Script for TradingView
//
// This script sends OHLCV data to a webhook endpoint for automatic
// collection and conversion to HDF5 format for backtesting.
//
// SETUP INSTRUCTIONS:
// 1. Open TradingView chart with MNQ futures (e.g., MNQM26)
// 2. Click "Pine Editor" at the bottom
// 3. Copy and paste this script
// 4. Click "Add to Chart" button
// 5. Click on the indicator settings (gear icon)
// 6. Go to the "Inputs" tab
// 7. Set your webhook URL (e.g., http://your-server.com:8080/webhook/tradingview)
// 8. Create an alert:
//    - Condition: MNQ Export Webhook
//    - Once Per Bar Close: YES (important!)
//    - Webhook URL: (same URL from step 7)
//    - Message: {{ticker}}, {{time}}, {{open}}, {{high}}, {{low}}, {{close}}, {{volume}}
//
// The webhook receiver will automatically:
// - Collect the data
// - Convert to HDF5 format
// - Save to data/historical/mnq/ directory
// - Auto-save every 500 bars

//@version=5
indicator("MNQ Export Webhook", overlay=false, max_labels_count=500)

// ===== CONFIGURATION =====
// Webhook URL (update this to match your server)
webhookUrl = input.string(
     "http://localhost:8080/webhook/tradingview",
     title="Webhook URL",
     tooltip="Your webhook receiver URL"
     )

// Symbol to use in filename
symbolName = input.string(
     "MNQM26",
     title="Symbol Name",
     tooltip="MNQ contract symbol for filename"
     )

// Enable/disable alerts
enableAlerts = input.bool(
     true,
     title="Enable Alerts",
     tooltip="Turn on to send data to webhook"
     )

// ===== DATA PREPARATION =====
// Format timestamp as ISO string
timestampStr = str.format("{0}-{1}-{2}T{3}:{4}:{5}Z",
     year, str.format("{0,number,00}", month), str.format("{0,number,00}", dayofmonth),
     str.format("{0,number,00}", hour), str.format("{0,number,00}", minute), str.format("{0,number,00}", second)
     )

// Create JSON payload
jsonData = str.format("{{\"symbol\":\"{0}\",\"timestamp\":\"{1}\",\"open\":{2},\"high\":{3},\"low\":{4},\"close\":{5},\"volume\":{6}}}",
     symbolName,
     timestampStr,
     str.tostring(open, "#.##"),
     str.tostring(high, "#.##"),
     str.tostring(low, "#.##"),
     str.tostring(close, "#.##"),
     str.tostring(volume, "#")
     )

// ===== ALERT CONDITION =====
// Trigger on every bar close
sendData = enableAlerts and barstate.isconfirmed

// ===== VISUAL FEEDBACK =====
// Show label when data is sent
if sendData and barstate.islast
    label.new(bar_index, high, "✓ Sending...", style=label.style_label_up, color=color.green, size=size.tiny)

// ===== ALERT =====
alertcondition(
     sendData,
     title="MNQ Export Webhook",
     message=jsonData
     )

// ===== INFO TABLE =====
// Display status on chart
var table statusTable = table.new(
     position.bottom_right,
     columns=2,
     rows=4,
     bgcolor=color.new(color.white, 90),
     border_width=1,
     border_color=color.gray
     )

if barstate.islast
    // Update status table
    table.cell(statusTable, 0, 0, "Status:", text_color=color.white, text_size=size.small)
    table.cell(statusTable, 1, 0, enableAlerts ? "Sending..." : "Disabled", text_color=enableAlerts ? color.green : color.red, text_size=size.small)

    table.cell(statusTable, 0, 1, "Webhook:", text_color=color.white, text_size=size.small)
    table.cell(statusTable, 1, 1, "Configured", text_color=color.green, text_size=size.small)

    table.cell(statusTable, 0, 2, "Symbol:", text_color=color.white, text_size=size.small)
    table.cell(statusTable, 1, 2, symbolName, text_color=color.white, text_size=size.small)

    table.cell(statusTable, 0, 3, "Last Bar:", text_color=color.white, text_size=size.small)
    table.cell(statusTable, 1, 3, str.format("{0}", bar_index), text_color=color.white, text_size=size.small)


// ===== USAGE NOTES =====
//
// This script will send data on every bar close when alerts are enabled.
//
// IMPORTANT:
// - Set "Once Per Bar Close" to YES in alert settings
// - Make sure your webhook server is running before enabling
// - Data is sent in JSON format compatible with the receiver
// - Receiver auto-saves every 500 bars to HDF5
//
// TROUBLESHOOTING:
// - No data received? Check webhook URL is correct
// - Missing bars? Check "Once Per Bar Close" is enabled
// - Server error? Check receiver script is running
//
// For manual export (without webhook), use this alternative script:
// See: tradingview_manual_export.txt
